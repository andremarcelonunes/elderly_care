@startuml
class User {
    + id: Integer
    + name: String
    + email: String
    + phone: String
    + role: String  // elderly, contact, nurse
    + password_hash: String
}

class Patient {
    + id: Integer
    + user_id: Integer
    + medical_conditions: Text
}

class Emergency {
    + id: Integer
    + patient_id: Integer
    + timestamp: DateTime
    + location: String
    + status: String
    + details: JSON  // Stores symptoms, indicators, etc.
}

class NotificationConfig {
    + id: Integer
    + patient_id: Integer
    + notify_contacts: Boolean
    + notify_nurses: Boolean
}

class Contact {
    + id: Integer
    + patient_id: Integer
    + contact_id: Integer
}
